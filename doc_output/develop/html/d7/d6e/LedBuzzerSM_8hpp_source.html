214  template<class Event, class Fsm, class TargetState>
215  void operator()(Event const &, Fsm &fsm, PlayingPattern &ss, TargetState &) const
216  {
217  if (ss.pattern_.size() == 0)
218  {
219  fsm.process_event(EventPlayingPatternStop());
220  return;
221  }
222  std::pair<int, int> current_pattern = ss.pattern_.back();
223  ss.pattern_.pop_back();
224 
225  // prepare blinking event, then fire it
226  EventBlink b;
227  b.duration = current_pattern.first;
228  b.speed = current_pattern.second;
229  fsm.process_event(b);
230  }
231 
235  template<class Event, class Fsm, class TargetState>
236  void operator()(Event const &, Fsm &fsm, StateBlinking &ss, TargetState &) const
237  {
238  fsm.gpio_.toggle();
239  fsm.next_update_time_ = std::chrono::system_clock::now() + std::chrono::milliseconds(ss.speed_);
240  if (--ss.nb_itr_ == 0)
241  {
242  fsm.process_event(EventBlinkStop());
243  }
244  }
